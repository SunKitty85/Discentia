drop database if exists discentia;
create database discentia;
use discentia;

create table tb_cards  (
id int auto_increment,
question text,
answer1 text,
answer2 text,
answer3 text,
answer4 text,
releaseDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
img_path varchar(50),
PRIMARY KEY (id));

create table tb_cards_category (
card_id int,
category_id int,
PRIMARY KEY (card_id,category_id));

create table tb_cards_subject (
card_id int,
subject_id int,
PRIMARY KEY (card_id,subject_id));

create table tb_pulls (
id int auto_increment,
polldatetime bigint,
user_ident bigint,
PRIMARY KEY (id));

create table tb_subject (
id int auto_increment,
subject varchar(50),
img_path varchar(50),
PRIMARY KEY (id));

CREATE TABLE tb_category (
id int auto_increment,
category varchar(50),
img_path varchar(50),
PRIMARY KEY (id));
